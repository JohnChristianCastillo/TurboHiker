294  // initialize the lookahead our AI has;
295  std::vector<std::shared_ptr<GlobalBounds>> simpleAILookAhead = simpleAi->getLookAhead();
296  for (int j = 0; j <= 2; ++j) {

324  void collisionWithEnemyControl(const std::shared_ptr<GlobalBounds>& nextPosition,
325  const std::shared_ptr<GlobalBounds>& aiNextPosition)
326  {
327  for (int i = enemies.size() - 1; i >= 0; --i) {
328 
329  // Try to prevent AI from colliding in the future
330  aiCollisionPrevention(enemies[i], nextPosition);
331 
332  // The MC's area of effect is the whole screen
333  // we need to know if our MC is scaring off enemy players, if so we add scarePoints to the
334  // player This in turn causes the enemy to panic steer this scarePoint has a threshold and once
335  // exceeded it can throw off the enemy
336  if (mainCharacter->isScaringEnemy() and mainCharacter->getScareCooldown() <= 0) {
337  if (enemies[i]->exceededScareThreshold()) {
338  enemies.erase(enemies.begin() + i);
339  scoringSystem->hikerThrownOff();
340  continue;
341  } else {
342  // if it hasn't exceeded the threshold yet, we will steer the enemy randomly
343  enemies[i]->setSteerRandolmy(true);
344  scoringSystem->hikerOffended();
345  }
346  }
347 
348  if (enemies[i]->getGlobalBounds()->intersects<float>(nextPosition)) {
349  // if we collided we want to penalize our player by slowing it down and subtracting
350  // points unless our main character is invincible
351  if (!mainCharacter->isInvincible()) {
352  scoringSystem->crashed();
353  mainCharacter->slowDown();
354  }
355 
356  enemies.erase(enemies.begin() + i);
357  }
358 
359  if (enemies[i]->getGlobalBounds()->intersects<float>(aiNextPosition)) {
360  // if we collided we want to penalize our player by slowing it down and subtracting
361  // points
362  simpleAi->slowDown();
363  enemies.erase(enemies.begin() + i);
364  }
365  }
